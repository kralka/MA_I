Spočítejte objem nekonečného \uv{trychtýře} vzniklého rotací $f(x) = 1/x$ na intervalu $x \in [1, \infty]$ okolo osy $x$.

\solution{
	Dle Věty~\ref{thm:objem_rotacniho_telesa} spočítáme
	\begin{align*}
		V &= \pi \int_{1}^{\infty} \left( 1/x \right)^2 \dx \\
		&= \pi \left[ -1/x \right]_{1}^{\infty} \\
		&= \pi
	\end{align*}
}
