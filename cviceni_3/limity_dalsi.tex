Určete limitu následujících posloupností $(a_n)_{n=1}^{\infty}$ nebo dokažte, že neexistují:
\begin{enumerate}
	\item  $a_n = (-1)^n$
	\item  $a_n = \cos((-1)^n)$
	\item  $a_n = (-1)^{n!}$
	\item  $a_n = \frac{(-1)^n}{n}$
	\item  $a_n = \cos(\frac{n \pi}{4})$
	\item  $a_n = \frac{2^n}{n^n}$
\end{enumerate}
