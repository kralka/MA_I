Pro dané množiny najděte jejich minimum, maximum, supremum, infimum (pokud existují).
\begin{enumerate}

	\item  $\left\{ 2^{-n} + 3^{-m} \mid n, m \in \mathbb{N} \right\}$

		\solution{
			Množina: $\left\{ \frac{5}{6}, \frac{11}{18}, \frac{7}{12}, \frac{29}{54}, \frac{83}{162}, \frac{245}{486}, \frac{731}{1458}, \frac{2189}{4374}, \frac{6563}{13122}, \ldots\right\}$.
			\begin{itemize}
					\item  $\min$ neexistuje
					\item  $\max = 5/6$
					\item  $\sup = 5/6$
					\item  $\inf = 0$
			\end{itemize}
		}

	\item  $\left\{ 5^{(-1)^j 3^k} \mid j, k \in \mathbb{Z} \right\}$

		\solution{
			\begin{itemize}
					\item  $\min$ neexistuje
					\item  $\max$ neexistuje
					\item  $\sup = \infty$ nebo můžeme říct, že v reálných číslech supremum nemá
					\item  $\inf = 0$
			\end{itemize}
		}

	\item  $\left\{ \cos((n + \frac{1}{n}) \pi) \mid n \in \mathbb{N} \right\}$
		
		\solution{
			Množina: $\left\{ 1, 0.9238, 0.8660, 0.7071, 0, -0.4999, \ldots \right\}$ (zaokrouhlené).
			\begin{itemize}
					\item  $\min$ neexistuje
					\item  $\max = 1$
					\item  $\sup = 1$
					\item  $\inf = -1$
			\end{itemize}
		}

	\item  $\left\{ \cos((n + \frac{1}{n}) \pi) \mid n \in \mathbb{N} \text{ sudé} \right\}$

		\solution{
			Množina: $\left\{ 0, 0.7071, 0.8660, 0.9238, 0.9510, \ldots \right\}$ (zaokrouhlené).
			\begin{itemize}
					\item  $\min = 0$
					\item  $\max$ neexistuje
					\item  $\sup = 1$
					\item  $\inf = 0$
			\end{itemize}
		}

\end{enumerate}
