\begin{definition}
	Nechť $f \colon M \rightarrow \mathbb{R}$, $b \in M$, $U(b, \delta) \subseteq M$ pro nějaké $\delta > 0$.
	Derivace funkce $f$ v bodě $b$ je limita:
	$$f'(b) = \lim_{h \rightarrow 0} \frac{f(b + h) - f(b)}{h}$$
	\label{def:derivace}
\end{definition}
